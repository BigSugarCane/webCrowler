   <script type="text/javascript">
	//中国用户注册获取手机验证码
	$(".passwd-sms").click(function () {
		var url = "index3.php?option=com_verifycode&task=show&act=send_verifycode().tb";
		TB_show("", url);
	});
	var timer = null;
	var seconds = 60;
	function couontDown() {
		seconds--;
		if (seconds == 0) {
			window.clearInterval(timer);
			$("#send_reset").attr("disabled", false);
			$("#send_reset").val("获取手机验证码");
			if ($("#vcodeMemo").length) {
				$("#vcodeMemo").hide();
			}
			seconds = 60;
		} else {
			$("#send_reset").val("重新发送" + seconds + "s");
		}
	}

	function send_verifycode(){
		var verify_code = $("#verify_code").val();
		TB_remove();
		if(!verify_code) return -1;
		var e = $(".input-phone");
		var u = $(".username");
		var _memo = $("#vcodeMemo");
		_memo.hide();
		//调用接口向手机发送验证码
		var flage = false;
		var phcode = "";
		$.ajax({
			type: "GET",
			url: "/index3.php",
			dataType: "json",
			async: false,
			cache: false,
			data: {
				type: "passwd",
				option: "com_bindtel",
				task: "retrieve",
				phone: e.val(),
				username: u.val(),
				verify_code: verify_code,
				random: Math.random()
			},
			success: function (data) {
				var _memo = $("#vcodeMemo");
				phcode = data.code;
				if (data.code == "200") {
					flage = true;
				} else if (data.code == "501") {
					_memo.text("参数有误！"),_memo.show();
				} else if (data.code == "502") {
					_memo.text("手机号码格式错误！"),_memo.show();
				} else if (data.code == "514") {
					_memo.text("该手机号码未绑定任何账号！"), _memo.show();
				} else if (data.code == "515") {
					_memo.text("用户不存在！"), _memo.show();
				} else if (data.code == "516") {
					_memo.text("用户未激活！"), _memo.show();
				} else if (data.code == "517") {
					_memo.text("用户名与手机号不匹配！"), _memo.show();
				} else if (data.code == "518") {
					_memo.text("校验码错误，请重新输入！"), _memo.show();
				} else if (data.code == "3") {
					_memo.text("请您稍等片刻后重新发送！"), _memo.show();
				} else {
					_memo.text("发送验证码失败！"), _memo.show();
				}
				if (data.code != "200") {
					return !1;
				}
			},
			error: function () {
			}
		});
		if (flage == true || phcode == '3') {
			$("#send_reset").attr("disabled", true);
			$("#send_reset").val("重新发送60s");
			timer = window.setInterval(couontDown, 1000);
		}
	}
    function reload_verifycode(){
        $("#varify_code_img_c").attr('src',"/index3.php?option=com_verifycode&task=load");
    }
    $('input[name="newpass_submit"]').click(function () {
        var verify_code = $("#verify_code_c").val();
        if(!verify_code) {
            alert("请输入图片上的计算结果");
            return -1;
        };
        $.ajax({
            type: "GET",
            url: "/index3.php",
            dataType: "json",
            async: false,
            cache: false,
            data: {
                type: "scode",
                option: "com_bindtel",
                task:'getConfirmCode',
                verify_code: verify_code
            },
            success: function (data) {
                if(data.code == "518"){
                    alert("计算结果输入错误，请重新输入！");
                    $("#varify_code_img_c").attr('src',"/index3.php?option=com_verifycode&task=load");
                    return -1;
                }else{
                    return true;
                }
            },
            error: function () {
            }
        });
    });
