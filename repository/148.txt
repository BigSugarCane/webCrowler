    function prefixInteger(num, length) {
        return (Array(length).join('0') + num).substr(-length);
    }

    function ClockData(dt) {
        var dt_year = dt.getUTCFullYear();
        var dt_month = dt.getUTCMonth() + 1;
        var dt_day = dt.getUTCDate();
        var dt_hour = dt.getUTCHours();
        var dt_minute = dt.getUTCMinutes();
        var dt_second = dt.getUTCSeconds();
        dt_year = dt_year.toString();
        dt_minute = prefixInteger(dt_minute, 2);
        dt_second = prefixInteger(dt_second, 2);
        return {
            year: dt_year,
            month: dt_month,
            day: dt_day,
            hour: dt_hour,
            minute: dt_minute,
            second: dt_second
        };
    }

    function ClockString(dt, dst) {
        var time = ClockData(dt);
        return (dst ? "*" : "") + time.year + '年' + time.month + '月' + time.day + '日' + ' ' +
            time.hour + ':' + time.minute + ':' + time.second;
    }

     <h2>佛罗里达时间</h2>
    </div> 
    <div class="clear"></div> 
    <div>
      佛罗里达的时区属于美国东部时间(EST)（西五区时间）,和美国西部重镇旧金山,圣地亚哥,洛杉矶相差三个时区.佛罗里达时间和中国的北京时间相差为13个小时,也就是中国的北京时间要比佛罗里达早13个小时,如果佛罗里达使用夏令时间,则中国北京时间比佛罗里达时间早12个小时.
